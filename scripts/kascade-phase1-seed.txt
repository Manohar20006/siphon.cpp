    void * device_ptr;
    std::uint64_t last_access;

class ExpertCache {
public:
    bool ensure_loaded(int layer, int expert, cudaStream_t stream) {
        const Key key{layer, expert};
        auto it = entries_.find(key);
        if (it != entries_.end()) {
            it->second.last_access = ++clock_;
            return true;
        }

        CacheEntry * victim = choose_victim();
        if (victim == nullptr) {
            return false;
        }

        if (!submit_gds_read(key, victim->device_ptr, stream)) {
            return false;
        }

        entries_.erase(Key{victim->layer, victim->expert});
        *victim = CacheEntry{layer, expert, expert_size(layer, expert),
                             victim->device_ptr, ++clock_};
        entries_.emplace(key, *victim);
        return true;
    }

private:
    struct Key {
        int layer;
        int expert;
        bool operator==(const Key &) const = default;
    };

    CacheEntry * choose_victim();
    bool submit_gds_read(const Key &, void *, cudaStream_t);
    std::size_t expert_size(int, int) const;

    std::unordered_map<Key, CacheEntry> entries_;
    std::uint64_t clock_ = 0;
};
